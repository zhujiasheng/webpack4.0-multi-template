
#### type、class、instance
```
type First = {
  name: string
  age: number
}

class Second2 {
  name: string
  age: number
}

interface Thrid {
  name: string
  age: number
}

let a: First = {
  name: 'rise',
  age: 11,
}

let b: Second2 = {
  name: 'rise2',
  age: 12,
}

let c: Thrid = {
  name: 'rise3',
  age: 13,
}
```
